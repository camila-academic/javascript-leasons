class Animal {
    constructor(voice){
        this.voice = voice || 'grunt';
    }

    speak() {
        display(this.voice)
    }
}

class Cat extends Animal {
    constructor(name, color) {
        super('Meow');
        this.name = name;
        this.color = color;
    }
}

var fluffy = new Cat('Fluffy', 'white');

console.log(Object.keys(fluffy));

const alunos = [
    { nome: "Carlos", notas: [7, 8, 9] },
    { nome: "Ana", notas: [5, 6, 4] }
];

function calcularMedia(notas) {
    const soma = notas.reduce((total, nota) => {
        return total + nota;
    }, 0);

    return soma / notas.length;
}

alunos.forEach(aluno => {
    const media = calcularMedia(aluno.notas);

    console.log(`\nAluno: ${aluno.nome}`);
    console.log(`Média: ${media.toFixed(2)}`);

    if (media >= 7) console.log("Status: Aprovado");
    else console.log("Status: Reprovado");
});
